#include <iostream> //头文件
using namespace std;//命名空间
int main()          //主函数  
{
	int a,b;        //int定义a和b
	cin>>a>>b;      //输入a和b
	cout<<a+b<<endl;//利用加号运算符将a和b加在一起并输出、换行
	return 0;       //主函数返回0（表示程序正常运行结束）
}
